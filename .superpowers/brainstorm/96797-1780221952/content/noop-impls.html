Phase 1 Design — Section 3: No-op Implementations

每个接口一个 no-op 实现。编译期安全，运行时零开销（内联为空）。消灭所有 nil 检查。

需要 No-op 的接口

// internal/agent/noop.go
type discardEmitter struct{}
func (discardEmitter) EmitSessionStart(_ string, _ string) {}
func (discardEmitter) EmitToolStart(_, _, _) {}
func (discardEmitter) EmitToolDone(_, _, _, _) {}
func (discardEmitter) EmitPhaseChange(_, _) {}
func (discardEmitter) EmitContextCompress(_, _, _, _, _) {}
func (discardEmitter) EmitMetricsUpdate(_, _, _, _, _, _, _, _, _) {}

// internal/tool/noop.go
type passthroughChain struct{}
func (passthroughChain) Execute(ctx, call, final) (*ToolResult, error) {
  return final(ctx, call) // 直接执行，无拦截
}

// internal/agent/noop_compressor.go
type noopCompressor struct{}
func (noopCompressor) Compress(ctx, sess, prompt) error { return nil }
func (noopCompressor) Utilization(sess, prompt) float64 { return 0 }

// internal/memory/noop.go
type noopStore struct{}
func (noopStore) Save(ctx, e) error { return nil }
func (noopStore) Search(ctx, q) ([]SearchResult, error) { return nil, nil }
func (noopStore) ListByScope(...) ([]Entry, error) { return nil, nil }
func (noopStore) Update(...) error { return nil }
func (noopStore) Delete(...) error { return nil }

// internal/knowledge/noop.go
type noopSearcher struct{}
func (noopSearcher) Search(ctx, q) ([]Result, error) { return nil, nil }

// internal/session/noop.go
type noopTaskLedger struct{}
func (noopTaskLedger) Register(ctx, t) error { return nil }
func (noopTaskLedger) Update(ctx, t) error { return nil }
func (noopTaskLedger) List(ctx, f) ([]Task, error) { return nil, nil }

使用对比

❌ Before: nil 散弹
if r.dashEmitter != nil {
  r.dashEmitter.EmitSessionStart(...)
}
if r.memStore != nil {
  memories, err := r.memStore.Search(...)
}
if r.interceptorChain != nil {
  result, _ := r.interceptorChain.Execute(...)
}
// ... scattered through EVERY method
✅ After: 直接调用
// deps.Observability.Emitter is NEVER nil
r.deps.Observability.Emitter.EmitSessionStart(...)

// deps.Memory.Store is NEVER nil
memories, _ := r.deps.Memory.Store.Search(...)

// deps.Security.Interceptor is NEVER nil
result, _ := r.deps.Security.Interceptor.Execute(...)

// ZERO nil checks. Compiler guarantees safety.
MemoryDeps.WithDefaults() 工厂方法
func (d MemoryDeps) WithDefaults() MemoryDeps {
  if d.Store == nil { d.Store = noopStore{} }
  if d.Compressor == nil { d.Compressor = noopCompressor{} }
  return d // LifecycleMgr/Profiler/FactExtractor stay nil (intentional)
}

// Gateway 构造处：一次补全，终身安全
deps := agent.AgentDeps{
  Core: agent.CoreDeps{...},
  Memory: agent.MemoryDeps{
    Store: gw.memory.Store(), // or nil
    LifecycleMgr: gw.memory.LifecycleMgr(),
  }.WithDefaults(), // ← fills noopStore{} if Store was nil
}
