// internal/agent/agent.go — ~150 lines
type Agent struct {
  deps AgentDeps
  strategy LoopStrategy
  approvalFn ApprovalFunc

}

func NewAgent(deps AgentDeps, strategy LoopStrategy, bus EventBus) *Agent

// HandleMessage — single entry point, all modes
func (a *Agent) HandleMessage(ctx, ch, msg) error {
  a.eventBus.Publish(SessionStarted{...})
  err := a.strategy.Execute(ctx, a, ch, msg)
  a.eventBus.Publish(SessionEnded{...})
  return err
}

3. EventBus — typed events, pub/sub

// internal/agent/events.go
type Event interface { EventType() string }

// 12 event types replace the 12 DashboardEmitter methods:
type SessionStarted struct { SessionID, Channel string }
type SessionEnded struct { SessionID string; Succeeded bool; DurationMs int64 }
type ToolExecuted struct { SessionID, ToolName string; Succeeded bool; DurationMs int64 }
type PhaseChanged struct { SessionID, Phase string; IsStart bool }
type ContextCompressed struct { SessionID, Reason string; BeforePct, AfterPct float64 }
type PlanGenerated struct { SessionID string; TaskCount int; Complexity string }
type ReplanStarted struct { SessionID string; Attempt int; Reason string }
type ObservationResult struct { SessionID string; Passed, Failed, Total int }
type SubAgentSpawned struct { SessionID, ParentID, AgentName, Task string }
type SubAgentDone struct { SessionID, AgentName string; Succeeded bool }
type MetricsTick struct { SessionID string; Iteration, MaxIter int; Utilization float64 }

type EventBus interface {
  Publish(event Event)
  Subscribe(fn func(Event)) Subscription
}

type Subscription interface { Unsubscribe() }

// inproc_bus.go — simple goroutine-safe implementation
type InprocBus struct {
  mu sync.RWMutex
  subs []func(Event)
  pending int64 // atomic counter for non-blocking publish
}

// Non-blocking publish: slow subscribers get dropped (already in dashboard/eventbus.go)
// Zero allocation fast path for no subscribers.

5. Impact

Benefits

  • Agent 唯一入口，不再 switch mode
  • 新循环策略 = 新 struct，不改 Agent
  • 事件总线替换 12 方法接口 + MultiEmitter 样板代码
  • dashboard/TUI/evolution 三个消费者统一订阅模式
  • runtime.go(884) + cognitive.go(1712) → agent.go(150) + simple_loop.go(350) + cognitive_loop.go(500)

Risks

  • 删除 2600 行核心代码，新增 ~1400 行 — 逻辑搬运必须精确
  • SubAgentManager 内部创建 Agent 实例，需要适配
  • 事件顺序敏感：PhaseChanged(PLAN, true) 必须在 PlanGenerated 之前
