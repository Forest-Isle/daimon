Phase 1 Design — Section 1: AgentDeps Structure

五个子 struct，每个有明确的 ownership 和默认 no-op

CoreDeps — 必须提供（无 no-op）

// internal/agent/deps.go
type CoreDeps struct {
  Provider Provider
  Tools *tool.Registry
  Sessions *session.Manager
  DB *store.DB
  Cfg config.AgentConfig
  LLMCfg config.LLMConfig
  AgentID string
}

// NewRuntime(deps CoreDeps) *Runtime ← 3个参数变1个
// NewCognitiveAgent(deps CoreDeps) *CognitiveAgent

MemoryDeps — 默认 no-op

// 所有字段默认 noopMemoryStore{}, noopLifecycleMgr{} 等
type MemoryDeps struct {
  Store memory.Store
  LifecycleMgr *memory.LifecycleManager // nil = NOOP decisions
  Profiler *memory.Profiler // nil = no profile updates
  Compressor Compressor // noopCompressor{}
  FactExtractor *memory.LLMFactExtractor // nil = no extraction
  BaseDir string
}
// 提供了 MemoryDeps{}.WithDefaults() 方法补全 no-op

SecurityDeps + ObservabilityDeps + MultiAgentDeps

type SecurityDeps struct {
  Interceptor *tool.InterceptorChain // nil = passthrough
  HookMgr *hook.Manager // nil = no hooks
  PermEngine *tool.PermissionEngine // nil = allow-all
}

type ObservabilityDeps struct {
  Emitter DashboardEmitter // discardEmitter{}
  MetricsEmitter MetricsEmitter // discardMetrics{}
  ReplayRecorder *ReplayRecorder // nil = no recording
}

type MultiAgentDeps struct {
  SkillMgr *skill.Manager
  AgentMgr *AgentManager
  Orchestrator *AgentOrchestrator
  SubAgentMgr *SubAgentManager // nil = no sub-agents
  AgentMCP *AgentMCPManager
  ResultStore *tool.ResultStore
}

// 完整 AgentDeps
type AgentDeps struct {
  Core CoreDeps
  Memory MemoryDeps
  Security SecurityDeps
  Observability ObservabilityDeps
  MultiAgent MultiAgentDeps
}

变更影响

  • 删除 ~30 个 Set* 方法（runtime.go）
  • 删除 ~20 个 Set* 方法（cognitive.go）
  • 删除 ~200 行 nil 检查散落代码
  • Runtime 从 68 字段 → 1 个 deps + 5 个自身状态字段
  • gateway 构造 1 个 AgentDeps，runtime 和 cognitiveAgent 共享

破坏性

  • 所有 NewRuntime / NewCognitiveAgent 调用点需改
  • 所有测试的 mock 构造方式需改
  • SubAgentManager 的依赖传递需适配
  • eval harness 的 Agent 构造需改
